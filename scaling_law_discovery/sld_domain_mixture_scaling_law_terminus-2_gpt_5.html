def _predict_one(sample: Dict[str, float], group: str) -> Dict[str, float]:
    if group not in PARAMS:
        raise ValueError(f"Unknown group: {group}")
    out: Dict[str, float] = {}
    gparams = PARAMS[group]
    for key, coeffs in gparams.items():
        # key looks like 'domain_i' from fitting stage
        d_idx = ''.join(ch for ch in key if ch.isdigit())
        p_key = f'proportion_domain_{d_idx}'
        y_key = f'loss_domain_{d_idx}'
        p = float(sample.get(p_key, 0.0))
        a = coeffs["a"]; b = coeffs["b"]; alpha = coeffs["alpha"]
        x = (p + EPS) ** (-alpha) if alpha != 0.0 else p
        y = a + b * x
        out[y_key] = float(y)
    return out

def _predict_domain(p: float, spec: dict) -> float:
    # Guard against non-positive proportions for log and power
    p = max(p, 1e-12)
    form = spec.get('form')
    if form == 'power':
        return spec['a'] + spec['k'] * (p ** spec['c'])
    elif form == 'log':
        return spec['a'] + spec['b'] * math.log(p)
    else:
        # Should not happen; fall back to identity-like
        return float('nan')

def law(input_data: list[dict[str, float]], group: str) -> list[dict[str, float]]:

    Predicts output variables based on input variables according to a discovered scaling law.

    Args:

        group: group name. Functional form is same across groups (per-domain chosen from {power, log}); coefficients differ per group.

    Returns:
        list of dicts with keys loss_domain_1..5

    # Select group-specific params or fallback
