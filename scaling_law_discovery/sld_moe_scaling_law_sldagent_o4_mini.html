"""
Log-exponential scaling law for MoE vs. dense transformer validation loss.
Model: 
  L(n, p) = C + exp( a0 
                     - alpha * log(n) 
                     - beta  * log(p/1e8) 
                     + gamma * log(n) * log(p/1e8) )
Params: [a0, alpha, beta, gamma, C] (5 total)
Fitting: joint nonlinear optimization (L-BFGS-B) minimizing MSE.
"""
import numpy as np
from scipy.optimize import minimize

def scaling_law_func(data_points, params):
    """
    data_points: (N,2) array [num_experts, dense_param_count]
    params: [a0, alpha, beta, gamma, C]
    returns: (N,) predicted validation loss
    """
    X = np.atleast_2d(data_points).astype(float)
    n       = X[:, 0]
    p_norm  = X[:, 1] / 1e8
    a0, alpha, beta, gamma, C = params

    # safeguard logs
    z1 = np.log(n + 1e-12)
    z2 = np.log(p_norm + 1e-12)
    exponent = a0 - alpha * z1 - beta * z2 + gamma * (z1 * z2)
    exponent = np.clip(exponent, -50, 50)  # prevent overflow
    return C + np.exp(exponent)

def fit_scaling_law(data_points, loss_values):
    """
    Fits the 5-parameter log-exponential scaling law:
      L = C + exp(a0 - α·log(n) - β·log(p_norm) + γ·log(n)·log(p_norm))
    using L-BFGS-B to minimize MSE. Returns [a0, alpha, beta, gamma, C].
    """
    X = np.atleast_2d(data_points).astype(float)
    y = np.asarray(loss_values, float).ravel()
    n      = X[:, 0]
    p_norm = X[:, 1] / 1e8

    # Precompute logs
    z1 = np.log(n + 1e-12)
    z2 = np.log(p_norm + 1e-12)

    # Initial guesses
    C0      = np.min(y) * 0.9
    a0      = np.log(np.maximum(np.mean(y - C0), 1e-6))
    alpha0  = 0.5
    beta0   = 0.5
    gamma0  = 0.0
    p0       = np.array([a0, alpha0, beta0, gamma0, C0])

    # MSE objective
    def mse(params):
        pred = params[4] + np.exp(
            params[0]
            - params[1] * z1
            - params[2] * z2
            + params[3] * (z1 * z2)
        )
        return np.mean((pred - y) ** 2)

    # Bounds for stability
    bounds = [
        (-10.0, 10.0),   # a0
        (1e-6, 5.0),     # alpha
        (1e-6, 5.0),     # beta
        (-1.0, 1.0),     # gamma
        (None, None)     # C
    ]

    res = minimize(
        mse,
        p0,
        method='L-BFGS-B',
        bounds=bounds,
        options={'ftol': 1e-12, 'maxiter': 2000}
    )

    return res.x if res.success else p0

"""
6-parameter saturating mixing scaling law for MoE:
  loss ≈ C + (A·(D/1e8)^α + B) / (E^γ + (D/1e8)^δ)
Parameters: [A, α, B, γ, δ, C]
"""
import numpy as np
from scipy.optimize import minimize

def scaling_law_func(data_points, params):
    X = np.atleast_2d(np.asarray(data_points, dtype=float))
    E = X[:, 0]                   # num_experts
    D = X[:, 1]                   # dense parameter count
    d = D / 1e8                   # normalized dense scale
    p = np.asarray(params, dtype=float)
    if p.ndim == 1:
        p = p[None, :]
    assert p.shape[1] == 6, "Expected 6 params [A, α, B, γ, δ, C]"
    A, alpha, B, gamma, delta, C = p.T

    num = A[None, :] * (d[:, None] ** alpha[None, :]) + B[None, :]
    den = (E[:, None] ** gamma[None, :]) + (d[:, None] ** delta[None, :])
    pred = C[None, :] + num / den
    return pred[:, 0] if pred.shape[1] == 1 else pred

def fit_scaling_law(data_points, loss_values):
    X = np.atleast_2d(np.asarray(data_points, dtype=float))
    y = np.asarray(loss_values, dtype=float).ravel()
    E = X[:, 0]
    D = X[:, 1]
    d = D / 1e8

    def objective(raw):
        logA, alpha, logB, logg, delta, C = raw
        A = np.exp(logA)
        B = np.exp(logB)
        gamma = np.exp(logg)
        num = A * (d ** alpha) + B
        den = (E ** gamma) + (d ** delta)
        pred = C + num / den
        return np.mean((pred - y) ** 2)

    # Initialize in log-space for scale parameters
    init = np.array([
        0.0,        # logA -> A≈1
        1.0,        # α  (dense exponent)
        0.0,        # logB -> B≈1
        0.0,        # logg-> γ≈1
        1.0,        # δ  (dense exponent in denom)
        np.mean(y)  # C  (baseline loss)
    ])
    bounds = [
        (-10, 10),  # logA
        (-5, 5),    # α
        (-10, 10),  # logB
        (-5, 5),    # logγ
        (-5, 5),    # δ
        (None, None)# C
    ]

    res = minimize(objective, init, method='L-BFGS-B', bounds=bounds)
    raw_opt = res.x if res.success else init
    logA, alpha, logB, logg, delta, C = raw_opt
    return np.array([
        np.exp(logA),
        alpha,
        np.exp(logB),
        np.exp(logg),
        delta,
        C
    ])

"""
Enhanced scaling law for MoE and dense transformer validation loss.
Model: L = A * N_experts^{-α} + B * (P/1e8)^{-β} 
          + D * [N_experts^{-α} * (P/1e8)^{-β}] + C
6 parameters: A, B, D, α, β, C
Fitting: optimize α,β via L-BFGS-B; solve A,B,D,C in closed-form least squares.
"""
import numpy as np
from scipy.optimize import minimize

def scaling_law_func(data_points, params):
    """
    data_points: (N,2) array [num_experts, dense_param_count]
    params: [A, B, D, alpha, beta, C]
    returns: (N,) predicted loss
    """
    X = np.atleast_2d(data_points)
    x1 = X[:,0].astype(float)
    x2 = X[:,1].astype(float) / 1e8
    A, B, D, alpha, beta, C = params
    f1 = x1**(-alpha)
    f2 = x2**(-beta)
    return A*f1 + B*f2 + D*(f1*f2) + C

def fit_scaling_law(data_points, loss_values):
    """
    Fits the 6-parameter scaling law by:
      - optimizing (alpha,beta) with L-BFGS-B
      - for each (alpha,beta), solves linear least squares for (A,B,D,C)
    Returns: params = [A, B, D, alpha, beta, C]
    """
    X = np.atleast_2d(data_points)
    y = np.asarray(loss_values).ravel()
    x1 = X[:,0].astype(float)
    x2 = X[:,1].astype(float) / 1e8
    N = len(y)

    def objective(ab):
        alpha, beta = ab
        # build design matrix for linear terms
        f1 = x1**(-alpha)
        f2 = x2**(-beta)
        f3 = f1 * f2
        Dmat = np.vstack((f1, f2, f3, np.ones(N))).T
        # solve for A,B,D_coef,C
        coefs, *_ = np.linalg.lstsq(Dmat, y, rcond=None)
        y_pred = Dmat.dot(coefs)
        return np.mean((y_pred - y)**2)

    # bounds to keep exponents positive and reasonable
    bnds = ((1e-8, 5.0), (1e-8, 5.0))
    init_ab = np.array([0.5, 0.5])
    res = minimize(objective, init_ab, bounds=bnds, method='L-BFGS-B')
    alpha_opt, beta_opt = res.x if res.success else init_ab

    # final linear solve for A,B,D_coef,C with best exponents
    f1 = x1**(-alpha_opt)
    f2 = x2**(-beta_opt)
    f3 = f1 * f2
    Dmat = np.vstack((f1, f2, f3, np.ones(N))).T
    coefs, *_ = np.linalg.lstsq(Dmat, y, rcond=None)
    A_opt, B_opt, D_opt, C_opt = coefs

    return np.array([A_opt, B_opt, D_opt, alpha_opt, beta_opt, C_opt])
