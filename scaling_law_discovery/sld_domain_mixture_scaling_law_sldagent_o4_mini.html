import numpy as np

def scaling_law_func(data_points, params):
    """
    Predict 5-domain losses from mixture proportions using 7 features per domain
    (35 parameters total):
      1) bias
      2) log(p_j + ε)
      3) p_j
      4) sqrt(p_j + ε)
      5) p_j * (1 - p_j)
      6) mixture entropy H
      7) H^2
    This replaces the p_j^2 feature with sqrt(p_j) for better mid‐range modeling
    and retains entropy‐based cross‐domain coupling.
    """
    X = np.asarray(data_points, dtype=float)
    N, D = X.shape
    assert D == 5, "Expected input shape (N,5)"
    
    p = np.asarray(params, dtype=float).ravel()
    assert p.size == 35, "Parameter vector must have length 35"
    W = p.reshape(5, 7)  # weights: one row per domain, one column per feature
    
    eps = 1e-8
    # feature precomputations
    logX   = np.log(X + eps)                             # (N,5)
    H      = -np.sum(X * logX, axis=1, keepdims=True)    # (N,1) entropy
    H2     = H * H                                       # (N,1)
    sqrtX  = np.sqrt(X + eps)                            # (N,5)
    crossX = X * (1.0 - X)                                # (N,5) p_j*(1-p_j)
    
    # build predictions
    preds = np.zeros((N, 5), dtype=float)
    for j in range(5):
        # slice out domain‐specific features
        pj_log   = logX[:,   j:j+1]  # (N,1)
        pj       = X[:,      j:j+1]  # (N,1)
        pj_sqrt  = sqrtX[:,   j:j+1]  # (N,1)
        pj_cross = crossX[:,    j:j+1]  # (N,1)
        
        # assemble feature matrix F_j: (N,7)
        F = np.hstack([
            np.ones((N, 1)),    # bias
            pj_log,             # log p_j
            pj,                 # p_j
            pj_sqrt,            # sqrt(p_j)
            pj_cross,           # p_j*(1-p_j)
            H,                  # entropy
            H2                  # entropy^2
        ])
        preds[:, j] = F.dot(W[j])
    
    return preds


def fit_scaling_law(data_points, loss_values):
    """
    Fit the 35 parameters by per‐domain ridge regression in closed form.
    Uses the same feature set as in scaling_law_func.
    """
    X = np.asarray(data_points, dtype=float)
    y = np.asarray(loss_values, dtype=float)
    N, D = X.shape
    assert D == 5, "Expected input shape (N,5)"
    
    # ensure y is (N,5)
    if y.ndim == 1:
        y = y[:, None]
    assert y.shape == (N, 5), "Expected loss_values shape (N,5)"
    
    eps = 1e-8
    # regularization (reduced slightly for tighter fit)
    reg = 1e-4
    
    # precompute shared features
    logX   = np.log(X + eps)                             # (N,5)
    H      = -np.sum(X * logX, axis=1, keepdims=True)    # (N,1)
    H2     = H * H                                       # (N,1)
    sqrtX  = np.sqrt(X + eps)                            # (N,5)
    crossX = X * (1.0 - X)                                # (N,5)
    
    # allocate weights
    W = np.zeros((5, 7), dtype=float)
    I7 = np.eye(7, dtype=float)
    
    # solve per domain
    for j in range(5):
        pj_log   = logX[:,   j:j+1]  # (N,1)
        pj       = X[:,      j:j+1]  # (N,1)
        pj_sqrt  = sqrtX[:,   j:j+1]  # (N,1)
        pj_cross = crossX[:,    j:j+1]  # (N,1)
        
        F = np.hstack([
            np.ones((N, 1)),    # bias
            pj_log,             # log p_j
            pj,                 # p_j
            pj_sqrt,            # sqrt(p_j)
            pj_cross,           # p_j*(1-p_j)
            H,                  # entropy
            H2                  # entropy^2
        ])  # (N,7)
        
        # ridge regression: (F^T F + reg I) w = F^T y
        A = F.T.dot(F) + reg * I7
        b = F.T.dot(y[:, j])
        W[j] = np.linalg.solve(A, b)
    
    return W.ravel()  # length 35

import numpy as np

def scaling_law_func(data_points, params):
    """
    Predict multi-domain losses from 5-d mixture proportions using:
      y_j = sum_i W[j,i] * p_i  +  gamma[j] * sqrt(p_j)  +  b[j]
    where we pack parameters as:
      - W: 5×5 weight matrix       (25 params)
      - gamma: 5-d vector          ( 5 params)
      - b: 5-d bias                ( 5 params)
    Total = 35 parameters.
    """
    X = np.asarray(data_points, dtype=float)
    N, F = X.shape
    assert F == 5, "Expected 5 mixture proportions"

    p = np.asarray(params, dtype=float).ravel()
    assert p.size == 35, f"Expected 35 parameters, got {p.size}"

    # unpack
    W_lin = p[:25].reshape(5, 5)
    gamma = p[25:30]
    b     = p[30:35]

    # linear part
    lin = X.dot(W_lin.T)               # (N,5)
    # non-linear sqrt effect per domain
    sqrt_eff = np.sqrt(X) * gamma      # broadcast to (N,5)

    return lin + sqrt_eff + b

def fit_scaling_law(data_points, loss_values):
    """
    Fit the above 35-param model with a two-stage procedure:
      1) initialize by small‐ridge per-domain least squares
      2) refine each domain via IRLS with Huber-like weights
    This yields robustness to outliers and tighter fit.
    """
    X = np.asarray(data_points, dtype=float)
    y = np.asarray(loss_values, dtype=float)
    N, F = X.shape
    assert F == 5, "Expected 5 mixture proportions"
    if y.ndim == 1:
        y = y[:, None]
    assert y.shape == (N, 5), "Expected loss_values shape (N,5)"

    # parameters to estimate
    W_lin = np.zeros((5, 5), dtype=float)
    gamma = np.zeros(5, dtype=float)
    b     = np.zeros(5, dtype=float)

    # ridge factor for stability
    lam = 1e-6 * N
    # IRLS iterations
    max_iter = 5

    # fit each domain separately
    for j in range(5):
        # build design: [p1..p5, sqrt(p_j), 1]
        sqrt_col = np.sqrt(X[:, j:j+1])               # (N,1)
        Phi = np.hstack((X, sqrt_col, np.ones((N, 1))))  # (N,7)
        yj  = y[:, j]

        # 1) initial ridge LS
        G = Phi.T.dot(Phi) + lam * np.eye(Phi.shape[1])
        v = Phi.T.dot(yj)
        theta = np.linalg.solve(G, v)

        # 2) IRLS refinement (Huber‐style weights)
        for _ in range(max_iter):
            resid = Phi.dot(theta) - yj
            med   = np.median(resid)
            mad   = np.median(np.abs(resid - med)) + 1e-6
            k     = 1.345 * mad
            # Huber weights: w=1 if |r|<=k else k/|r|
            absr = np.abs(resid)
            w    = np.where(absr <= k, 1.0, k / absr)

            # weighted normal equations
            Wphi = Phi * w[:, None]
            Gw   = Wphi.T.dot(Phi) + lam * np.eye(Phi.shape[1])
            vw   = Wphi.T.dot(yj)
            theta = np.linalg.solve(Gw, vw)

        # unpack domain params
        W_lin[j, :] = theta[:5]
        gamma[j]    = theta[5]
        b[j]        = theta[6]

    # flatten into length‐35 vector
    return np.concatenate((W_lin.ravel(), gamma, b))

def scaling_law_func(data_points, params):
    """
    Predict multi-domain losses (5 outputs) from 5-domain mixture proportions
    using a shared‐exponent power transform + per‐domain linear weights + bias.
    Total params = 5 alphas + 5×5 weights + 5 biases = 35.
    """
    X = np.asarray(data_points, dtype=float)
    if X.ndim == 1:
        X = X[None, :]
    N, F = X.shape
    assert F == 5, f"Expected 5 mixture proportions, got {F}"
    p = params.ravel()
    # unpack
    alphas = p[0:F]                          # (5,)
    W_flat = p[F:F + F*F]                   # (25,)
    bias   = p[F + F*F:F + F*F + F]         # (5,)
    W = W_flat.reshape(5, 5)                # (5 outputs × 5 features)
    # safe power transform (handles zeros via small epsilon)
    eps = 1e-8
    G = np.exp(np.log(X + eps) * alphas)    # (N,5)
    # linear mixing + bias
    preds = G.dot(W.T) + bias[None, :]
    return preds

def fit_scaling_law(data_points, loss_values):
    """
    Fit the 35 parameters by minimizing a regularized MSE over all 5 domains.
    Returns optimized flat parameter vector of length 35.
    """
    X = np.asarray(data_points, dtype=float)
    y = np.asarray(loss_values, dtype=float)
    if X.ndim == 1:
        X = X[None, :]
    if y.ndim == 1:
        y = y[None, :]
    N, F = X.shape
    T = y.shape[1]
    assert F == 5 and T == 5, "Expected 5 mixture proportions and 5 loss dims"
    # initialize: alphas=1, W=identity, bias=mean per-domain loss
    alphas0 = np.ones(F, dtype=float)
    W0 = np.eye(T, F, dtype=float)
    bias0 = np.mean(y, axis=0)
    init_params = np.concatenate([alphas0, W0.ravel(), bias0])
    # bounds: alphas in [0,5], weights in [-5,5], bias free
    bounds = [(0.0, 5.0)] * F + [(-5.0, 5.0)] * (F*F) + [(None, None)] * T
    # small L2 regularization for stability
    reg_alpha = 1e-3
    reg_W     = 1e-3

    def objective(p):
        pred = scaling_law_func(X, p)
        mse  = np.mean((pred - y)**2)
        a = p[:F]
        w = p[F:F + F*F]
        # encourage alphas ≈1 and weights ≈0
        penalty = reg_alpha * np.sum((a - 1.0)**2) + reg_W * np.sum(w**2)
        return mse + penalty

    res = minimize(
        objective,
        init_params,
        method='L-BFGS-B',
        bounds=bounds,
        options={'maxiter': 2000, 'ftol': 1e-9}
    )
    return res.x if res.success else init_params

import numpy as np

def scaling_law_func(data_points, params):
    """
    Quadratic-per-domain scaling law:
      For each domain j:
        pred[:,j] = X @ a_j + (X[:,j]**2)*d_j + b_j
    where a_j is a 5-vector, d_j is a scalar weight on the j-th squared input,
    and b_j is a bias.  Total parameters = 5*(5 + 1 + 1) = 35.
    """
    X = np.atleast_2d(data_points).astype(float)
    N, D = X.shape
    assert D == 5, "Expected 5 domain proportions"
    p = params.ravel()
    assert p.size == 35, "Expected 35 parameters"
    pred = np.zeros((N, 5), dtype=float)
    # unpack and compute per-domain prediction
    for j in range(5):
        base = j * 7
        a_j = p[base:base+5]       # linear weights for domain j
        d_j = p[base+5]            # squared-term weight for domain j
        b_j = p[base+6]            # bias for domain j
        pred[:, j] = X.dot(a_j) + (X[:, j]**2) * d_j + b_j
    return pred

def fit_scaling_law(data_points, loss_values):
    """
    Fit the quadratic-per-domain model via ridge‐regularized least squares.
    For each domain j, design matrix is [X, X[:,j]**2, 1] of shape (N,7).
    Solve (D^T D + λI) θ = D^T y for θ of length 7, then pack into params.
    """
    X = np.atleast_2d(data_points).astype(float)
    Y = np.atleast_2d(loss_values).astype(float)
    N, D = X.shape
    assert D == 5 and Y.shape == (N, 5)
    X2 = X * X
    lam = 1e-6
    params = np.zeros(35, dtype=float)
    for j in range(5):
        # build per-domain design matrix (N,7)
        Xj2 = X2[:, j:j+1]
        Dmat = np.hstack((X, Xj2, np.ones((N, 1), dtype=float)))
        yj = Y[:, j]
        # normal equations with ridge on all but bias
        G = Dmat.T.dot(Dmat)
        for k in range(6):
            G[k, k] += lam
        rhs = Dmat.T.dot(yj)
        theta = np.linalg.solve(G, rhs)
        # pack parameters
        base = j * 7
        params[base:base+5] = theta[:5]
        params[base+5]      = theta[5]
        params[base+6]      = theta[6]
    return params
