def scaling_law_func(data_points, params):
    """
    A simplified two-component scaling law:
      Lossu ≈ (C0 * P^{-a} + C1 * V^{-g}) * D^{-b} + C2
    where
      P = non-vocab parameter count,
      V = vocabulary size,
      D = number of characters seen.
    params = [C0, C1, C2, a, b, g]  (6 parameters)
    """
    X = np.atleast_2d(data_points)
    P = X[:, 0]
    V = X[:, 1]
    D = X[:, 2]
    C0, C1, C2, a, b, g = params
    # share the same data‐exponent b for both P and V contributions
    return (C0 * P**(-a) + C1 * V**(-g)) * (D**(-b)) + C2

def fit_scaling_law(data_points, loss_values):
    """
    Fit the 6-parameter model by:
      1) optimizing exponents (a, b, g) in log-space via L-BFGS-B,
         while solving C0, C1, C2 by linear least squares inside the objective,
      2) selecting the best of multiple restarts,
      3) recovering the final linear coefficients with fixed exponents.
    Returns params = [C0, C1, C2, a, b, g].
    """
    X = np.atleast_2d(data_points)
    P = X[:, 0]
    V = X[:, 1]
    D = X[:, 2]
    y = np.asarray(loss_values).ravel()

    def _mse(log_exps):
        a, b, g = np.exp(log_exps)
        D_term = D**(-b)
        M = np.vstack([
            P**(-a) * D_term,
            V**(-g) * D_term,
            np.ones_like(P)
        ]).T
        coefs, *_ = np.linalg.lstsq(M, y, rcond=None)
        y_pred = M.dot(coefs)
        return np.mean((y - y_pred)**2)

    # bounds for exponents in log-space
    bounds = [(np.log(1e-8), np.log(10.0))] * 3
    best_mse = np.inf
    best_log = None
    # multi-start to avoid local minima
    for init in ([0.5, 0.5, 0.5], [0.3, 0.8, 0.4], [1.0, 0.2, 0.6]):
        res = minimize(_mse, np.log(init),
                       method='L-BFGS-B', bounds=bounds)
        if res.fun < best_mse:
            best_mse, best_log = res.fun, res.x

    a_opt, b_opt, g_opt = np.exp(best_log)
    # now solve for C0, C1, C2 with fixed exponents
    D_term = D**(-b_opt)
    M_final = np.vstack([
        P**(-a_opt) * D_term,
        V**(-g_opt) * D_term,
        np.ones_like(P)
    ]).T
    C0_opt, C1_opt, C2_opt = np.linalg.lstsq(M_final, y, rcond=None)[0]

    return np.array([C0_opt, C1_opt, C2_opt, a_opt, b_opt, g_opt])

def scaling_law_func(data_points, params):
    """
    Combined multiplicative and data-only power‐law scaling:
      Lossu ≈ C0
             + C1 * P^(−alpha) * D^(−beta) * V^(−gamma)
             + C2 * D^(−delta)

    params = [C0, C1, C2, alpha, beta, gamma, delta]
      C0     : bias / asymptotic loss
      C1     : amplitude of full P·D·V interaction term
      C2     : amplitude of data‐only term
      alpha  : exponent on non‐vocab parameters P
      beta   : exponent on number of chars D in interaction
      gamma  : exponent on vocab size V
      delta  : exponent on number of chars D in data‐only term
    """
    X = np.atleast_2d(data_points).astype(float)
    P = X[:, 0]
    V = X[:, 1]
    D = X[:, 2]
    C0, C1, C2, alpha, beta, gamma, delta = params

    # compute with logs for numerical stability
    logP = np.log(P)
    logD = np.log(D)
    logV = np.log(V)

    f_main = np.exp(-alpha * logP - beta * logD - gamma * logV)
    f_data = np.exp(-delta * logD)
    return C0 + C1 * f_main + C2 * f_data

def fit_scaling_law(data_points, loss_values):
    """
    Fit the 7‐parameter model
      Lossu ≈ C0
             + C1 * P^(−alpha) * D^(−beta) * V^(−gamma)
             + C2 * D^(−delta)

    by:
      1) optimizing exponents (alpha, beta, gamma, delta) in log‐space
      2) solving linear least squares for (C1, C2, C0)
    Returns np.array([C0, C1, C2, alpha, beta, gamma, delta]).
    """
    X = np.atleast_2d(data_points).astype(float)
    y = np.asarray(loss_values).ravel()
    P = X[:, 0]
    V = X[:, 1]
    D = X[:, 2]

    # precompute logs
    logP = np.log(P)
    logD = np.log(D)
    logV = np.log(V)

    def mse_exponents(log_exps):
        # enforce exponents > 0
        alpha, beta, gamma, delta = np.exp(log_exps)
        # build design matrix
        f_main = np.exp(-alpha * logP - beta * logD - gamma * logV)
        f_data = np.exp(-delta * logD)
        M = np.vstack((f_main, f_data, np.ones_like(y))).T
        # least squares for [C1, C2, C0]
        coeffs, *_ = np.linalg.lstsq(M, y, rcond=None)
        y_pred = M.dot(coeffs)
        return np.mean((y - y_pred) ** 2)

    # initialize exponents in log space
    init_log = np.log([0.5, 0.5, 0.5, 0.5])
    bounds = [(np.log(1e-6), np.log(1e2))] * 4
    res = minimize(mse_exponents, init_log, method='L-BFGS-B', bounds=bounds)

    # extract optimized exponents
    alpha_opt, beta_opt, gamma_opt, delta_opt = np.exp(res.x)

    # final linear fit for amplitudes and bias
    f_main_opt = np.exp(-alpha_opt * logP - beta_opt * logD - gamma_opt * logV)
    f_data_opt = np.exp(-delta_opt * logD)
    M_final = np.vstack((f_main_opt, f_data_opt, np.ones_like(y))).T
    coeffs_final, *_ = np.linalg.lstsq(M_final, y, rcond=None)
    C1_opt, C2_opt, C0_opt = coeffs_final

    return np.array([C0_opt, C1_opt, C2_opt,
                     alpha_opt, beta_opt, gamma_opt, delta_opt])

def scaling_law_func(data_points, params):
    """
    Predict unigram-normalized loss (Lossu) from:
      data_points: array of shape (N,3): [P_non_vocab, vocab_size, num_characters]
      params: [C0, C1, C2, C3, alpha, beta, gamma]
    Model:
      Lossu = C0
            + C1 * (P^{-alpha} * D^{-beta})
            + C2 * V^{-gamma}
            + C3 * (P^{-alpha} * D^{-beta} * V^{-gamma})
    """
    X = np.atleast_2d(data_points)
    P = X[:, 0]
    V = X[:, 1]
    D = X[:, 2]
    C0, C1, C2, C3, alpha, beta, gamma = params
    # Base PD and V terms
    t_pd = P**(-alpha) * D**(-beta)
    t_v  = V**(-gamma)
    # Combined interaction
    t_pdv = t_pd * t_v
    return C0 + C1 * t_pd + C2 * t_v + C3 * t_pdv

def fit_scaling_law(data_points, loss_values):
    """
    Fit the 7-parameter scaling law
      Lossu = C0
            + C1 * (P^{-alpha} * D^{-beta})
            + C2 * (V^{-gamma})
            + C3 * (P^{-alpha} * D^{-beta} * V^{-gamma})
    by alternating linear least-squares for {C0,C1,C2,C3}
    and L-BFGS-B over the log-exponents {alpha,beta,gamma}.
    Returns optimized params [C0, C1, C2, C3, alpha, beta, gamma].
    """
    X = np.atleast_2d(data_points)
    y = np.asarray(loss_values).ravel()
    P = X[:, 0]
    V = X[:, 1]
    D = X[:, 2]

    # objective over log-exponents
    def obj(log_exps):
        a, b, g = np.exp(log_exps)
        t_pd = P**(-a) * D**(-b)
        t_v  = V**(-g)
        # design: [1, t_pd, t_v, t_pd*t_v]
        M = np.vstack([np.ones_like(y), t_pd, t_v, t_pd * t_v]).T
        C, *_ = np.linalg.lstsq(M, y, rcond=None)
        pred = M.dot(C)
        return np.mean((pred - y)**2)

    # initialize exponents log-space
    init = np.log([0.5, 0.5, 0.5])
    res = minimize(obj, init, method='L-BFGS-B')
    a_opt, b_opt, g_opt = np.exp(res.x)

    # final linear solve for coefficients
    t_pd = P**(-a_opt) * D**(-b_opt)
    t_v  = V**(-g_opt)
    M_final = np.vstack([np.ones_like(y), t_pd, t_v, t_pd * t_v]).T
    C0_opt, C1_opt, C2_opt, C3_opt = np.linalg.lstsq(M_final, y, rcond=None)[0]

    return np.array([C0_opt, C1_opt, C2_opt, C3_opt, a_opt, b_opt, g_opt])

import numpy as np

# EVOLVE-BLOCK-START
def scaling_law_func(data_points, params):
    """
    Predict unigram-normalized loss (Lossu) using a 7-parameter linear
    model in transformed log-space features that capture both
    growth (via logs) and saturation (via inverse logs).
    
    Features:
      p = log(non_vocab_parameters)
      v = log(vocab_size)
      d = log(num_characters)
      ip = 1/p, iv = 1/v, id = 1/d
    
    Model: Lossu ≈ A*1 + B*p + C*d + D*v + E*(1/p) + F*(1/d) + G*(1/v)
    """
    X = np.atleast_2d(data_points)
    # log-transformed inputs
    p = np.log(X[:, 0])
    v = np.log(X[:, 1])
    d = np.log(X[:, 2])
    # safe inverse logs
    eps = 1e-8
    ip = 1.0 / np.where(np.abs(p) > eps, p, eps)
    iv = 1.0 / np.where(np.abs(v) > eps, v, eps)
    id = 1.0 / np.where(np.abs(d) > eps, d, eps)
    
    # design matrix: [1, p, d, v, ip, id, iv]
    F = np.column_stack((np.ones_like(p), p, d, v, ip, id, iv))
    return F.dot(np.asarray(params).reshape(-1))

def fit_scaling_law(data_points, loss_values):
    """
    Fit the 7-parameter model by ridge-regularized least squares
    on features [1, log P, log D, log V, 1/log P, 1/log D, 1/log V].
    Returns parameter array of length 7.
    """
    X = np.atleast_2d(data_points)
    y = np.asarray(loss_values).ravel()
    # log-transform
    p = np.log(X[:, 0])
    v = np.log(X[:, 1])
    d = np.log(X[:, 2])
    # safe inverse logs
    eps = 1e-8
    ip = 1.0 / np.where(np.abs(p) > eps, p, eps)
    iv = 1.0 / np.where(np.abs(v) > eps, v, eps)
    id = 1.0 / np.where(np.abs(d) > eps, d, eps)
    
    # build design
    M = np.column_stack((np.ones_like(p), p, d, v, ip, id, iv))
    # ridge regularization for stability
    reg = 1e-6
    MTM = M.T @ M
    MTM[np.diag_indices_from(MTM)] += reg
    # solve for params
    params = np.linalg.solve(MTM, M.T @ y)
    return params
