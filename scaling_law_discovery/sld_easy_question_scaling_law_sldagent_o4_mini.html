import numpy as np
from scipy.optimize import least_squares

def scaling_law_func(data_points, params):
    """
    U-shaped scaling law: a quadratic valley plus a smooth tanh tail shift.
    params = [a, b, c, d, e, f]
      a > 0 : curvature of the quadratic valley
      b     : abscissa of the valley minimum
      c     : baseline offset
      d     : amplitude of the late‐scale shift (tanh tail)
      e > 0 : steepness of the transition in the tail
      f     : center of the tail transition
    y(x) = a*(x - b)^2 + c + d * tanh( e*(x - f) )
    """
    x = np.asarray(data_points).reshape(-1)
    a, b, c, d, e, f = params
    z = e * (x - f)
    return a * (x - b)**2 + c + d * np.tanh(z)

def _jacobian(params, x):
    """
    Analytic Jacobian of scaling_law_func w.r.t. params at points x.
    """
    a, b, c, d, e, f = params
    dx = x - b
    z = e * (x - f)
    t = np.tanh(z)
    sech2 = 1.0 - t*t
    m = x.size
    J = np.empty((m, 6), dtype=float)
    # d/d a
    J[:, 0] = dx**2
    # d/d b
    J[:, 1] = -2.0 * a * dx
    # d/d c
    J[:, 2] = 1.0
    # d/d d
    J[:, 3] = t
    # d/d e
    J[:, 4] = d * dx * sech2
    # d/d f
    J[:, 5] = -d * e * sech2
    return J

def fit_scaling_law(data_points, loss_values):
    """
    Fit the 6-parameter U-shaped scaling law to (log_flops, brier_score) data.
    Returns optimized params [a, b, c, d, e, f].
    """
    x = np.asarray(data_points).reshape(-1)
    y = np.asarray(loss_values).reshape(-1)
    xmin, xmax = x.min(), x.max()
    # Heuristic initial guess
    span = max(y.max() - y.min(), 1e-6)
    a0 = span / ((xmax - xmin)**2 + 1e-6)
    b0 = x[np.argmin(y)]
    c0 = np.median(y)
    d0 = (np.min(y) - c0)
    e0 = 1.0
    f0 = x[np.argmin(y)]
    p0 = np.array([a0, b0, c0, d0, e0, f0], dtype=float)

    # Bounds: enforce positive curvature & steepness; centers within data range
    lower = [1e-8, xmin,   -np.inf, -np.inf, 1e-8, xmin]
    upper = [np.inf,  xmax,  np.inf,  np.inf, np.inf, xmax]

    # Residual & Jacobian
    def resid(p):
        return scaling_law_func(x, p) - y
    def jac(p):
        return _jacobian(p, x)

    # Robust non-linear least squares
    result = least_squares(
        resid,
        p0,
        jac=jac,
        bounds=(lower, upper),
        loss='soft_l1',
        f_scale=0.1,
        max_nfev=2000
    )
    return result.x if result.success else p0

import numpy as np
from scipy.optimize import least_squares
from scipy.special import expit

def scaling_law_func(data_points, params):
    """
    U-shaped scaling law: quadratic curvature + logistic tail adjustment.
    y(x) = a*(x - b)^2 + c + d / (1 + exp(-e*(x - f)))
    params: [a, b, c, d, e, f]
      a >= 0      controls U-shape curvature
      b ∈ [xmin,xmax]    center of the quadratic trough
      c ∈ ℝ       baseline at x=b
      d ∈ ℝ       amplitude of logistic tail (improvement plateau)
      e >= 0      steepness of logistic transition
      f ∈ [xmin,xmax]    center of logistic transition
    """
    X = np.atleast_2d(np.asarray(data_points))
    x = X[:, 0]
    a, b, c, d, e, f = np.ravel(params)
    # Quadratic U-shape
    quad = a * (x - b)**2
    # Stable logistic tail: d/(1 + exp(-e*(x - f)))
    z = -e * (x - f)
    z = np.clip(z, -50.0, 50.0)
    logi = d * (1.0 - expit(z))
    return quad + c + logi

def fit_scaling_law(data_points, loss_values):
    """
    Fit the six-parameter U-shaped scaling law via robust least-squares.
    Returns optimized params [a,b,c,d,e,f].
    """
    X = np.atleast_2d(np.asarray(data_points))
    x = X[:, 0]
    y = np.ravel(loss_values)

    # Data bounds
    xmin, xmax = x.min(), x.max()
    xm = np.median(x)

    # Initial estimates:
    # 1) Quadratic trough at median x
    b0 = xm
    # 2) Baseline c0 by local interpolation at xm
    c0 = np.interp(xm, x, y)
    # 3) Curvature a0 from variance estimate
    span = xmax - xmin + 1e-6
    a0 = max(1e-8, np.var(y) / (span**2))
    # 4) Tail amplitude d0 from mean on right minus baseline
    right_mask = x >= xm
    if right_mask.sum() >= 3:
        d0 = np.mean(y[right_mask]) - c0
    else:
        d0 = -abs(np.median(y)) * 0.1
    # 5) Logistic steepness e0
    e0 = 1.0
    # 6) Transition center f0 at median
    f0 = xm

    p0 = np.array([a0, b0, c0, d0, e0, f0], dtype=float)

    # Parameter bounds
    lower = [0.0,        xmin,   -np.inf,   -np.inf,  0.0,    xmin]
    upper = [np.inf,     xmax,    np.inf,    np.inf,  np.inf, xmax]

    def residuals(p):
        return scaling_law_func(X, p) - y

    # Robust fitting with Cauchy loss to mitigate outliers
    result = least_squares(
        residuals,
        p0,
        bounds=(lower, upper),
        loss='cauchy',
        f_scale=0.1,
        max_nfev=4000,
        xtol=1e-8,
        ftol=1e-8,
        gtol=1e-8
    )

    return result.x if result.success else p0

    """
    U-shaped law: convex quadratic minus a logistic “improvement” tail.
    params = [a, b, c, d, e, f]:
      a ≥ 0    curvature of U
      b        center of U
      c        baseline offset
      d ≥ 0    amplitude of the late‐scale improvement
      e ≥ 0    steepness of the improvement
      f        center of the improvement transition
    """
    x = np.asarray(data_points).ravel()
    a, b, c, d, e, f = np.ravel(params)
    quad = a * (x - b)**2
    tail = d / (1.0 + np.exp(-e * (x - f)))
    # subtract the tail so that performance first degrades then improves
    return quad + c - tail

    Fit the six‐parameter U‐shaped law

        return scaling_law_func(x, p) - y

    xmin, xmax = x.min(), x.max()

    # sensible initialization

import numpy as np
from scipy.optimize import least_squares

def scaling_law_func(data_points, params):
    """
    U-shaped law: quadratic baseline + logistic tail (6 params).
      y = a*(x-b)^2 + c + d/(1+exp(-e*(x-f)))
    where a,e >= 0, d <= 0.
    """
    x = np.ravel(data_points)
    a, b, c, d, e, f = params
    return a*(x - b)**2 + c + d/(1 + np.exp(-e*(x - f)))

def fit_scaling_law(data_points, loss_values):
    """
    Fit the 6-parameter U-shaped scaling law via a two-stage least-squares:
    1) robust soft-L1 to handle outliers
    2) final linear (L2) to refine the fit
    """
    x = np.ravel(data_points)
    y = np.ravel(loss_values)
    xmin, xmax = x.min(), x.max()
    span = max(xmax - xmin, 1.0)

    # Smart initial guesses
    a0 = np.ptp(y)/(span**2 + 1e-6)        # curvature ~ variation/span^2
    b0 = float(x[np.argmax(y)])           # center at worst performance
    c0 = np.median(y)                     # baseline
    d0 = min(np.min(y) - c0, 0.0)         # negative tail amplitude
    e0 = 1.0                              # logistic steepness
    f0 = b0                               # logistic midpoint at b0
    p0 = np.array([a0, b0, c0, d0, e0, f0])

    # Bounds: a,e >= 0; d <= 0; b,f within [xmin, xmax]
    lower = [0.0,      xmin, -np.inf, -np.inf, 0.0,    xmin]
    upper = [np.inf,   xmax,  np.inf,   0.0,   np.inf, xmax]

    # Stage 1: robust fit
    res1 = least_squares(
        lambda p: scaling_law_func(x, p) - y,
        p0,
        bounds=(lower, upper),
        loss='soft_l1',
        f_scale=1.0,
        max_nfev=2000
    )
    if not res1.success:
        return p0

    # Stage 2: refine with true L2 loss
    res2 = least_squares(
        lambda p: scaling_law_func(x, p) - y,
        res1.x,
        bounds=(lower, upper),
        loss='linear',
        max_nfev=2000
    )
    return res2.x if res2.success else res1.x

"""
U-shaped scaling law: quadratic valley minus a logistic-improvement tail.
Six parameters:
  a - curvature of the U (≥0)
  b - center of the quadratic valley (in log_flops)
  c - baseline offset
  d - amplitude of improvement tail (≥0)
  e - steepness of improvement logistic (≥0)
  f - center of the improvement logistic (in log_flops)
"""
import numpy as np
from scipy.optimize import least_squares

def scaling_law_func(data_points, params):
    """
    data_points: (N,1) array of log_flops
    params: [a,b,c,d,e,f]
    returns: (N,) predicted brier_score (negative is better)
    """
    X = np.atleast_2d(data_points)
    x = X[:,0]
    a, b, c, d, e, f = params
    quad = a * (x - b)**2
    tail = d / (1.0 + np.exp(-e * (x - f)))
    # subtract tail to model late improvement
    return quad + c - tail

def fit_scaling_law(data_points, loss_values):
    """
    Fit the 6-parameter U-shape minus logistic tail via robust least squares.
    Returns optimized [a,b,c,d,e,f].
    """
    X = np.atleast_2d(data_points)
    x = X[:,0]
    y = np.ravel(loss_values)
    xmin, xmax = x.min(), x.max()

    def resid(p):
        return scaling_law_func(X, p) - y

    # initial guesses
    span = xmax - xmin if xmax > xmin else 1.0
    a0 = (np.ptp(y) / (span**2)) if span>0 else 1.0
    b0 = np.median(x)
    c0 = np.median(y)
    d0 = max(0.0, c0 - np.min(y))
    e0 = 1.0
    f0 = xmin + 0.75 * span
    p0 = np.array([a0, b0, c0, d0, e0, f0])

    # bounds: enforce a,d,e ≥ 0, centers within [xmin,xmax]
    lower = [0.0, xmin, -np.inf, 0.0, 0.0, xmin]
    upper = [np.inf, xmax, np.inf, np.inf, np.inf, xmax]

    result = least_squares(
        resid,
        p0,
        bounds=(lower, upper),
        loss='huber',
        f_scale=0.1,
        max_nfev=1500
    )
    return result.x if result.success else p0
