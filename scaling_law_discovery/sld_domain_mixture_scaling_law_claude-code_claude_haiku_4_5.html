    """
    Predicts output variables based on input variables according to a discovered scaling law.

    The scaling law models the validation loss for each domain as a linear function of:
    1. The proportion of that domain in the training mixture
    2. The logarithm of the model size (in millions of parameters)

    Functional form: loss_domain_i = a_i * proportion_domain_i + b_i * log(M) + c_i
    where M is the model size in millions.

    Args:
        input_data: A list of dictionaries, where each dictionary is a single data
                    point containing input variable names as keys and their
                    corresponding values. Expected keys are 'proportion_domain_1'
                    through 'proportion_domain_5'.
        group: The name of the experimental group for which to make predictions.
               Must be one of: '70M', '160M', '305M', '410M'

    Returns:
        A list of dictionaries, corresponding to the input_data list, with each
        dictionary containing the predicted output variables 'loss_domain_1'
        through 'loss_domain_5'.
    """

    # Model size in millions for each group
    group_sizes = {
        '70M': 70,
        '160M': 160,
        '305M': 305,
        '410M': 410
    }

    # Fitted parameters: {domain: {'a': coef_prop, 'b': coef_log_model_size, 'c': intercept}}
    params = {
        1: {'a': -1.125872, 'b': -0.304742, 'c': 4.430192},

        4: {'a': -2.206028, 'b': -0.234356, 'c': 3.039213},
        5: {'a': -0.423390, 'b': -0.332244, 'c': 5.163817},
    }

    # Get model size for the group
    model_size = group_sizes[group]
    log_model_size = math.log(model_size)

    results = []

    for data_point in input_data:
        prediction = {}

        for domain in range(1, 6):
            prop_key = f'proportion_domain_{domain}'

            p = params[domain]

            prediction[loss_key] = predicted_loss

        results.append(prediction)

    return results
