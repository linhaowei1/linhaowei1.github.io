def law(input_data: List[Dict[str, float]], group: str) -> List[Dict[str, float]]:
    """
    Predicts output variables based on input variables according to a discovered scaling law.

    The discovered law models the language modeling loss as a sum of a floor term
    plus two diminishing-return power-law contributions: one from the total
    number of parameters and one from the degree of model parallelism.

    Form:
        loss = c0[group] + c1[group] * num_params**(-a) + c2[group] * parallel_size**(-b)

    where (a, b) are shared across groups and the coefficients (c0, c1, c2) are
    group-specific constants.

    Args:
        input_data: A list of dictionaries, where each dictionary is a single data
                    point containing input variable names as keys and their
                    corresponding values. Expected keys: 'num_params', 'parallel_size'.
        group: The name of the experimental group for which to make predictions.
               Supported groups: 'stack', 'pile'. If an unknown group is supplied,
               a conservative default is used (closest to the overall fit).

    Returns:
        A list of dictionaries, corresponding to the input_data list, with each
        dictionary containing the predicted output variable(s) under key 'loss'.
    """

    # Shared exponents (common functional form across groups)
    a = 0.26
    b = 0.49

    # Group-specific coefficients (fitted on the provided dataset)
    coeffs = {
        "stack": {"c0": 0.74683622, "c1": 66.36773787, "c2": 0.06734961},
        "pile": {"c0": 1.37803186, "c1": 116.559736, "c2": 0.103430375},
    }

    # Fallback: if unknown group, use coefficients closest to overall fit.
    params = coeffs.get(group)
    if params is None:
        # Default to a set of coefficients roughly in between the two groups.
        # This maintains functional validity while avoiding a hard failure.
        params = {"c0": 1.06, "c1": 90.0, "c2": 0.085}

    c0, c1, c2 = params["c0"], params["c1"], params["c2"]

    outputs: List[Dict[str, float]] = []
    for row in input_data:
        n = float(row.get("num_params", 0.0))
        p = float(row.get("parallel_size", 1.0))

        # Guard against degenerate inputs
        if n <= 0.0:
            raise ValueError("num_params must be positive")
        if p <= 0.0:
            raise ValueError("parallel_size must be positive")

        loss = c0 + c1 * (n ** (-a)) + c2 * (p ** (-b))
        outputs.append({"loss": float(loss)})

    return outputs

def _predict_loss(num_params: float, parallel_size: float, a: float, b: float, c: float) -> float:
    """Compute loss using a log-linear power-law form.

    loss = exp(a + b * ln(num_params) + c * ln(parallel_size))
    """
    # Guard against non-positive inputs to logs; fall back to minimal positive value.
    n = max(float(num_params), 1e-12)
    p = max(float(parallel_size), 1e-12)
    return float(math.exp(a + b * math.log(n) + c * math.log(p)))

from typing import Dict, List


def _predict_loss(num_params: float, parallel_size: float, coeffs: Dict[str, float]) -> float:
    L_inf = coeffs["L_inf"]
    k = coeffs["k"]
    alpha = coeffs["alpha"]
    gamma = coeffs["gamma"]
    # Scaling law: L = L_inf + k * (num_params * parallel_size**gamma) ** (-alpha)
    n_eff = float(num_params) * (float(parallel_size) ** float(gamma))
    return float(L_inf + k * (n_eff ** (-alpha)))


def law(input_data: List[Dict[str, float]], group: str) -> List[Dict[str, float]]:
    """
    Predicts output variables based on input variables according to a discovered scaling law.

    Args:
        input_data: A list of dictionaries, where each dictionary is a single data
                    point containing input variable names as keys and their
                    corresponding values.
        group: The name of the experimental group for which to make predictions.
                The functional form of the law must be the same for all groups,
                but the constant parameters/coefficients can differ per group.

    Returns:
        A list of dictionaries, corresponding to the input_data list, with each
        dictionary containing the predicted output variable(s).
    """

    # Fitted coefficients per group (derived from /app/data)
    GROUP_COEFFS: Dict[str, Dict[str, float]] = {
        # group: { L_inf, k, alpha, gamma }
        "pile": {"L_inf": 0.9994680998203908, "k": 14.076803963829533, "alpha": 0.1260671701767887, "gamma": 0.5},
        "stack": {"L_inf": 0.6239972998774463, "k": 11.721235202089247, "alpha": 0.15204205303795576, "gamma": 0.5},
    }

    # Default to an average of known groups if an unseen group is provided
    if group in GROUP_COEFFS:
        coeffs = GROUP_COEFFS[group]
    else:
        # Simple average of per-parameter values
        vals = list(GROUP_COEFFS.values())
        coeffs = {
            "L_inf": sum(v["L_inf"] for v in vals) / len(vals),
            "k": sum(v["k"] for v in vals) / len(vals),
            "alpha": sum(v["alpha"] for v in vals) / len(vals),
            "gamma": sum(v["gamma"] for v in vals) / len(vals),
        }

    outputs: List[Dict[str, float]] = []
    for row in input_data:
        n = float(row.get("num_params", 0.0))
        p = float(row.get("parallel_size", 1.0))
        pred_loss = _predict_loss(n, p, coeffs)
        outputs.append({"loss": pred_loss})

    return outputs
